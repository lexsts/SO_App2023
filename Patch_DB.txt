SELECT sys_context('USERENV','DB_NAME') AS INSTANCE,PRODUCT,STATUS EDITION,VERSION PRODUCT_LEVEL FROM PRODUCT_COMPONENT_VERSION WHERE UPPER(PRODUCT) LIKE '%ORACLE%';

SELECT
SERVERPROPERTY('servername') AS INSTANCIA,
CASE SUBSTRING(CAST(SERVERPROPERTY('ProductVersion') AS nvarchar),1,2)
WHEN '9.' THEN 'SQL Server 2005 ('+CAST(SERVERPROPERTY('ProductVersion') AS nvarchar)+')'
WHEN '10' THEN 'SQL Server 2008 ('+CAST(SERVERPROPERTY('ProductVersion') AS nvarchar)+')'
WHEN '11' THEN 'SQL Server 2012 ('+CAST(SERVERPROPERTY('ProductVersion') AS nvarchar)+')'
WHEN '12' THEN 'SQL Server 2014 ('+CAST(SERVERPROPERTY('ProductVersion') AS nvarchar)+')'
END PRODUCT,
SERVERPROPERTY('Edition') AS EDITION,
SERVERPROPERTY('ProductLevel') AS PRODUCT_LEVEL
GO
GO
